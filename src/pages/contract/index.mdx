export const ContractNotLoaded = () => {
	const { contractLoaded } = useContract();
	if (!contractLoaded) {
		return (
			<p
				style={{
					marginBottom: '0.8rem'
				}}
			>
				<span
					style={{
						borderStyle: 'solid',
						borderWidth: '2px',
						borderColor: '#ef4444',
						display: 'inline-block',
						borderRadius: '99px',
						fontSize: '14px',
						fontWeight: 'bold',
						color: '#ef4444',
						padding: '0.2rem 0.8rem',
					}}
				>
					No contract imported
				</span>
				<a
					href="/import"
					style={{
						marginLeft: '0.8rem',
						fontWeight: 'bold'
					}}
				>
					Import your ABI here first.
				</a>
			</p>
		)
	}
	return null;
};

export const ContractData = ({ data = null }) => {
	const { contract, events, functions } = useContract();
	if (!contract || !data) {
		return null;
	}
	switch (data) {
		case "filename":
			return contract[data] || "/";
			break;
		case "events":
			return events.length;
			break;
		case "functions":
			return functions.length;
			break;
		default:
			return null;
			break;
	}
	return null;
};

export const getTOC = (instances) => instances.map(instance => ({
	value: instance.name,
	id: instance.name,
	children: [],
	level: 3,
}));

export const ContractConstructor = () => {
	const { constructor } = useContract();
	if (!!constructor) {
		return (
			<div>
				<div className={styles.instance}>
					<h3 id={constructor.type}>
						<code className={styles.mutability}>{constructor.stateMutability}</code>
					</h3>
					<div>
						<h4>
							Parameters
						</h4>
						{!!constructor.inputs.length ? (
							<table>
								<thead>
									<tr align="left">
										<th>Name</th>
										<th>Type</th>
									</tr>
								</thead>
								<tbody>
									{constructor.inputs.map(input => (
										<tr key={input.name}>
											<td>
												<strong>
													{input.name}
												</strong>
											</td>
											<td>
												<code>
													{input.type}
												</code>
											</td>
										</tr>
									))}
								</tbody>
							</table>
						): (
							<p className={styles.emptyState}>No parameters found in the constructor.</p>
						)}
					</div>
				</div>
			</div>
		);
	}
	return (
		<p className={styles.emptyState}>No <strong>constructor</strong> found in this smart contract.</p>
	);
};

export const ContractFunctions = () => {

	if (!!functions.length) {
		return (
			<div>
				<TOCInline toc={getTOC(functions)} />
				<hr className={styles.hr} />
				{functions.map(fn => (
					<div key={fn.name} className={styles.instance}>
						<h3 id={fn.name}>
							<code className={styles.title}>{ fn.name }</code>
							<code className={styles.mutability}>{fn.stateMutability}</code>
						</h3>
						<div className={styles.tables}>
							<div>
								<h4>
									Parameters
								</h4>
								{!!fn.inputs.length ? (
									<table>
										<thead>
											<tr align="left">
												<th>Name</th>
												<th>Type</th>
											</tr>
										</thead>
										<tbody>
											{fn.inputs.map(input => (
												<tr key={input.name}>
													<td>
														<strong>
															{input.name || '/'}
														</strong>
													</td>
													<td>
														<code>
															{input.type}
														</code>
													</td>
												</tr>
											))}
										</tbody>

								) : (

export const ContractEvents = () => {

	}
	return (
